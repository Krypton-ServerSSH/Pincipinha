        const audioPlayer = new Audio();

        audioPlayer.loop = true;

        audioPlayer.preload = 'auto'; // Cache imediato


        function preloadAssets() {

            // Preload Imagens

            fotosIDs.forEach(id => {

                const img = new Image();

                img.src = `https://drive.google.com/thumbnail?id=${id}&sz=w1000`;

            });


            // Preload Áudio com listener de buffer

            audioPlayer.src = linkMusica;

            audioPlayer.load();

            

            audioPlayer.addEventListener('canplaythrough', () => {

                document.getElementById('loader').innerText = "Tudo pronto! ✨";

            }, { once: true });


            setTimeout(() => {

                const loader = document.getElementById('loader');

                if(loader) loader.style.opacity = '0';

            }, 4000);

        }


        function criarPolaroids() {

            const container = document.getElementById('polaroid-container');

            const cols = 4, rows = 4;

            const cellW = 100 / cols, cellH = 100 / rows;


            fotosIDs.forEach((id, index) => {

                const div = document.createElement('div');

                div.className = 'polaroid';

                const img = document.createElement('img');

                img.src = `https://drive.google.com/thumbnail?id=${id}&sz=w1000`;

                div.appendChild(img);

                

                const col = index % cols;

                const row = Math.floor(index / cols);

                const jitterX = (Math.random() * 15 - 7.5); 

                const jitterY = (Math.random() * 15 - 7.5);


                div.style.left = `${(col * cellW) + 5 + jitterX}%`;

                div.style.top = `${(row * cellH) + 5 + jitterY}%`;

                div.style.zIndex = index;

                div.style.setProperty('--rotation', (Math.random() * 40 - 20) + "deg");

                div.style.setProperty('--float-speed', (Math.random() * 2 + 3) + "s");

                div.style.animationDelay = (Math.random() * 2) + "s";


                container.appendChild(div);

                setTimeout(() => div.classList.add('aparecer'), index * 80);

            });

        }


        function abrirSurpresa() {

            // Play imediato aproveitando a interação do usuário

            const playPromise = audioPlayer.play();

            if (playPromise !== undefined) {

                playPromise.catch(() => {

                    // Fallback para navegadores muito restritivos

                    setTimeout(() => audioPlayer.play(), 100);

                });

            }

            });


            criarPolaroids();

                        });

                    });

                }, 100);

            }, 500);

        }


        function digitar(id, texto, velocidade, callback) {

            let i = 0;

            const element = document.getElementById(id);

            element.innerHTML = "";

            function step() {

                if (i < texto.length) {

                    element.innerHTML += texto.charAt(i) === '\n' ? '<br>' : texto.charAt(i);

                    i++;

                    setTimeout(step, velocidade);

                } else if (callback) callback();

            }

            step();

        }
